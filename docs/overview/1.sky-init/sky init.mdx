# init all
`sky init`

# sky.config.json
`sky init sky-config`

```typescript
export default {
    title: '',
    apps: {
        'app name': {
            path: '.',
            target: 'web'
        }
    },
    modules: {
        name: {
            path: 'module path'
        }
    },
    scripts: {
        some: "some command"
    }
}
```

# tsconfig
`sky init tsconfig`

# .gitignore
`sky init gitignore`

# package.json
`sky init package`

# packages and lint configs
`sky init packages`
